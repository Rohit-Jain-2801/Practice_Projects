# Cosine_Similarity => cos0 where 0 is the angle between two pts. => [-1, 1]
# Cosine_Distance => 1-cosine_similarity => [0, 2] => lower means higher similarity
dist = 1 - cosine_similarity(df)

pca = PCA(2)
X_PCA = pca.fit_transform(dist)
X_PCA.shape

x, y = X_PCA[:, 0], X_PCA[:, 1]

colors = {0: 'red',
          1: 'blue',
          2: 'green', 
          3: 'yellow', 
          4: 'orange',  
          5:'purple'}
dfa = np.array(df)
